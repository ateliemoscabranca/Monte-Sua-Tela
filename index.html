<script>


let dados={};


function select(el,campo,valor,nextId){

  el.parentNode.querySelectorAll('.card').forEach(c=>c.classList.remove('selected'));

  el.classList.add('selected');


  dados[campo]=valor;

  atualizarResumo();

  scrollToNext(nextId);

}


function scrollToNext(id){

  document.getElementById(id).scrollIntoView({behavior:'smooth'});

}


function previewImage(e){

  let img=document.getElementById("preview");

  img.src=URL.createObjectURL(e.target.files[0]);

  img.style.display="block";

}


function atualizarResumo(){

  document.getElementById("resumo").innerHTML=

  `Formato: ${dados.Formato || '-'}<br>

   Tamanho: ${dados.Tamanho || '-'}<br>

   Estilo: ${dados.Estilo || '-'}<br>

   Acabamento: ${dados.Acabamento || '-'}`;

}


function enviar(){

  let nome=document.getElementById("nome").value;

  let tel=document.getElementById("telefone").value;


  let msg=`Pedido:%0A

Formato:${dados.Formato}

Tamanho:${dados.Tamanho}

Estilo:${dados.Estilo}

Acabamento:${dados.Acabamento}

Cliente:${nome}

Tel:${tel}`;


  window.open(`https://wa.me/5531997969797?text=${msg}`);

}
